BREAKDOWN STRUKTUR ROUTE CAFEPOS BY DATABASE => CAFEPOS (Sistem Kasir Cafe/Restoran + Inventory + Laporan)

Controllers/Auth/ControllerAuthUser -> middleware/RoleMiddleware(owner)   -> /loginowner
Controllers/Auth/ControllerAuthUser -> middleware/RoleMiddleware(manager) -> /loginmanager
Controllers/Auth/ControllerAuthUser -> middleware/RoleMiddleware(kasir)   -> /loginkasir

[Middleware RoleMiddleware (owner + manager + kasir)]
└───*dashboard division
    ├──/dashboardowner (dashboard khusus owner)
    ├──/dashboardmanager (dashboard khusus manager)
    └──/dashboardkasir (dashboard khusus kasir)

└───*auth division
    └──/loginhistory (id,userid,ipaddress,useragent,loginat,logoutat,status)

└───*master division
    ├──/user (id,name,username,email,password,role(owner,manager,kasir),foto,isactive)
    ├──/kategori (id,namakategori,deskripsi)
    ├──/produk (id,kategoriid,kodeproduk,namaproduk,hargajual,satuan,foto,status(aktif,nonaktif))
    ├──/meja (id,nomormeja,kapasitas,status(kosong,terisi))
    ├──/supplier (id,namasupplier,nohp,alamat)
    ├──/metodepembayaran (id,namametode,jenis(cash,noncash),status(aktif,nonaktif))
    ├──/promo (id,namapromo,jenis(persen,fixed),nilaidiskon,minimalbelanja,tanggalmulai,tanggalselesai,status(aktif,nonaktif))
    └──/pajak (id,namapajak,persen,status(aktif,nonaktif))

└───*inventory division
    ├──/bahanbaku (id,kodebahan,namabahan,stok,satuan,hargabeli)
    ├──/stok (id,bahanbakuid,stoktersedia,stokminimal,status(aman,habis))
    ├──/stokmasuk (id,bahanbakuid,jumlah,tanggalmasuk,keterangan)
    ├──/stokkeluar (id,bahanbakuid,jumlah,tanggalkeluar,alasan)
    ├──/pembelian (id,kodepembelian,supplierid,userid,total,tanggalpembelian)
    └──/detailpembelian (id,pembelianid,bahanbakuid,qty,harga,subtotal)

└───*transaksi division (POS)
    ├──/penjualan (id,kodeinvoice,userid,shiftid,mejaid,promoid,pajakid,subtotal,diskon,pajak,total,status(pending,paid),tanggalpenjualan)
    ├──/detailpenjualan (id,penjualanid,produkid,qty,harga,subtotal)
    ├──/pembayaran (id,penjualanid,metodepembayaranid,jumlahbayar,kembalian,tanggalbayar,buktibayar,status(paid,unpaid))
    └──/cetakstruk (id,penjualanid,strukfile,tanggalcetak)

└───*shift division
    └──/shift (id,userid,shiftmulai,shiftselesai,saldoawal,saldoakhir,totaltransaksi,status(open,closed))

└───*laporan division (generate from transaksi + inventory)
    ├──/laporan (id,userid,jenislaporan,periodeawal,periodeakhir,totaldata,created_at)
    ├──/laporanharian (id,userid,tanggal,totaltransaksi,totalpendapatan,totaldiskon,totalpajak)
    ├──/laporanbulanan (id,userid,bulan,tahun,totaltransaksi,totalpendapatan,totaldiskon,totalpajak)
    ├──/laporanproduk (id,userid,produkid,totalterjual,totalpendapatan)
    ├──/laporankasir (id,userid,kasirid,tanggal,totaltransaksi,totalpendapatan)
    ├──/laporanshift (id,userid,shiftid,tanggal,totaltransaksi,totalpendapatan)
    └──/laporankeuntungan (id,userid,tanggal,totalpemasukan,totalpengeluaran,keuntungan)

└───*zonakasir division
    └──/zonakasir (id,userid,statusaktif,catatan)

└───*landing division (guest)
    ├──/home (id,title,content,statusaktif)
    ├──/menu (id,produkid,kategoriid,statusaktif)
